/* ===================================== 🦉 OWL EVOLUTION SYSTEM (STEP 13) DROP-IN FOR lesson.html ===================================== */ /** * Get owl state based on XP */ function getOwlState(){ let xp = Engine.getXP(); let level = Math.floor(xp / 50); if(level < 2) return "neutral"; if(level < 5) return "happy"; if(level < 10) return "strong"; return "master"; } /** * Update owl appearance */ function updateOwl(){ let owl = document.getElementById("owl"); if(!owl) return; let state = getOwlState(); // reset owl.className = ""; // evolve owl visually if(state === "neutral"){ owl.innerText = "🦉"; } if(state === "happy"){ owl.innerText = "🦉✨"; } if(state === "strong"){ owl.innerText = "🦉🔥"; } if(state === "master"){ owl.innerText = "🦉👑"; } } /** * Helper (safe call wrapper) */ function refreshOwl(){ updateOwl(); }
