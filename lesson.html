/* ===================================== 🦉 OWL EVOLUTION SYSTEM (STEP 13) ===================================== */ /** * Determines owl state based on XP level */ function getOwlState(){ let xp = Engine.getXP(); let level = Math.floor(xp / 50); if(level < 2) return "neutral"; if(level < 5) return "happy"; if(level < 10) return "strong"; return "master"; } /** * Updates owl emoji + visual state */ function updateOwl(){ let owl = document.getElementById("owl"); if(!owl) return; let state = getOwlState(); // reset owl.className = ""; // set visual evolution if(state === "neutral"){ owl.innerText = "🦉"; } if(state === "happy"){ owl.innerText = "🦉✨"; } if(state === "strong"){ owl.innerText = "🦉🔥"; } if(state === "master"){ owl.innerText = "🦉👑"; } } /** * OPTIONAL: call this after every lesson action */ function refreshOwl(){ updateOwl(); }
